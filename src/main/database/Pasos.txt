-- Meter datos en el orden sugerido
NOTA: Tener los archivos .csv en una carpeta con permisos (la carpeta C:)
1. Personas
COPY persona(cedula,nombres,apellidos,direccion,telefono,email) FROM '/persona.csv' DELIMITER ',' CSV HEADER;

2. Usuarios
COPY usuario(cedula) FROM '/usuario.csv' DELIMITER ',' CSV HEADER;

3. Area
COPY area(idArea,nombreA,descripcionA) FROM '/area.csv' DELIMITER ',' CSV HEADER;

4. Cargo
COPY cargo(idCargo,nombreC,descripcionC,idArea) FROM '/cargo.csv' DELIMITER ',' CSV HEADER;

5. Empleado
COPY empleado(cedula, idCargo) FROM '/empleado.csv' DELIMITER ',' CSV HEADER;

6. Habitacion
COPY habitacion(noHabitacion,categoria,precionoche,estado) FROM '/habitacion.csv' DELIMITER ',' CSV HEADER;

7. Reserva
COPY reserva(idReserva,fechaEntrada,fechaSalida,noHabitacion,cedula) FROM '/reserva.csv' DELIMITER ',' CSV HEADER;

8. Servicio
COPY servicio(idServicio,detalles,nombre,costo) FROM '/servicio.csv' DELIMITER ',' CSV HEADER;

9. Consume
COPY consume(fecha,idReserva,idServicio,idConsumo) FROM '/consume.csv' DELIMITER ',' CSV HEADER;
